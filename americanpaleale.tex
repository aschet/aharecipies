\begin{recipe}{Maplewood Brewing Co. Charlatan American Pale Ale Clone}
% -----------------------------------------------------------------------------

\begin{aboutblock}
Packed with tropical flavors like mango, passionfruit, and grapefruit. \sourceaha
\end{aboutblock}

\specifications{\styleamericanpaleale}{\galtol{5}}{1.059}{1.012}{6.1~\%}{32}{\srmtoebc{5.1}}{90~min}{2.55}

\begin{methodandtiming}
 
\begin{mashsteps}
\mashstep{\ftoc{152}}{45~min}
\end{mashsteps}

\begin{directions}
Water adjustment: \waterprofile{115}{--}{--}{145}{65}{}, target mash pH at
5.3. Add the whirlpool hops at \ftoc{190}. When the fermentation is about
\ptop{1} from full attenuation, add the dry hops. After 3 days after that,
cold crash and transfer off of the hops.
\end{directions}

\end{methodandtiming}

\recipebreak

\begin{ingredientsblock}

\begin{malts}
\malt{Rahr Standard Two-row}{\lbtokg{8}}
\malt{Weyermann Munich I}{\lbtokg{0.75}}
\malt{Crisp Dextrin}{\lbtokg{0.33}}
\malt{Crisp Cara}{\lbtokg{0.33}}
\malt{Rahr White Wheat}{\lbtokg{0.33}}
\end{malts}

\begin{hops}
\hop{\hopwarrior}{16.5~\%}{90~min}{\oztog{0.1}}
\hop{\hopsimcoe}{13~\%}{15~min}{\oztog{0.2}}
\hop{\hopcitra}{12.5~\%}{10~min}{\oztog{0.1}}
\hop{\hopcentennial}{10~\%}{10~min}{\oztog{0.1}}
\hop{\hopsimcoe}{13~\%}{10~min}{\oztog{0.1}}
\hop{\hopcitra}{12.5~\%}{\whirl{}{10~min}}{\oztog{0.33}}
\hop{\hopcentennial}{10~\%}{\whirl{}{10~min}}{\oztog{0.33}}
\hop{\hopsimcoe}{13~\%}{\whirl{}{10~min}}{\oztog{0.165}}
\hop{\hopcitra}{12.5~\%}{\dryh{}{3~days}}{\oztog{1.25}}
\hop{\hopcentennial}{10~\%}{\dryh{}{3~days}}{\oztog{0.75}}
\hop{\hopsimcoe}{13~\%}{\dryh{}{3~days}}{\oztog{0.75}}
\end{hops}

\singleyeast{Wyeast 1318}

\end{ingredientsblock}

\end{recipe}
