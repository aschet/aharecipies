\begin{recipe}{Rogue Chocolate Stout Clone} % rechecked
% -----------------------------------------------------------------------------

\begin{aboutblock}
Smooth, sweet and velvety, and tastes like a chocolate malted milkshake with
slight hoppiness and bittersweet chocolate notes. \sourcezymurgy{July / August 2009}
\end{aboutblock}

\specifications{\styleamericanstout}{\galtol{5}}{1.069}{1.017}{6.8~\%}{30}{\srmtoebc{25}}{90~min}{}

\begin{methodandtiming}
 
\begin{mashsteps}
\mashstep{\ftoc{150}}{60~min}
\end{mashsteps}

\begin{fermentationsteps}
\fermentationstep{\ftoc{60}}{7~days}
\fermentationstep{\ftoc{50}}{Transfer to secondary; full attenuation}
\end{fermentationsteps}

\begin{directions}
Transfer to secondary at \ftoc{50} onto chocolate extract and hold until
fermentation is complete, then package and condition.
\end{directions}

\end{methodandtiming}

\recipebreak

\begin{ingredientsblock}

\begin{malts}
\malt{Great Western Premium Two-row}{\lbtokg{11}}
\malt{Caramel / Crystal 120 L}{\lbtokg{0.5}}
\malt{Chocolate}{\lbtokg{0.5}}
\malt{Rolled Oats}{\lbtokg{0.5}}
\malt{Roasted Barley}{\oztokg{3}}
\end{malts}

\begin{hops}
\hop{\hopcascade}{5~\%}{90~min}{\oztog{1}}
\hop{\hopcascade}{5~\%}{30~min}{\oztog{1}}
\hop{\hopcascade}{5~\%}{\foh{}}{\oztog{1}}
\end{hops}

\singleyeast{Wyeast 1764-PC / White Labs WLP001}

\begin{twists}
\twist{Chocolate Extract}{Secondary}{\oztog{1.51}}
\end{twists}

\end{ingredientsblock}

\end{recipe}
