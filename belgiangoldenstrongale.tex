\begin{recipe}{Nuetnigenough Belgian Golden Strong}
% -----------------------------------------------------------------------------

\begin{aboutblock}
Recipe by Wesley Carmichael \& Mandy Naglich of New York, NY. Silver medal in
Category 22: Strong Belgian Ale during the 2019 National Homebrew Competition
in Providence, RI. \sourceaha
\end{aboutblock}

\specifications{\stylebelgiangoldenstrongale}{\galtol{6}}{1.071}{1.010}{8.9~\%}{27}{\srmtoebc{3.3}}{90~min}{3}

\begin{methodandtiming}
 
\begin{mashsteps}
\mashstep{\ftoc{154}}{30~min}
\mashstep{\ftoc{170}}{Raise over 4~min; 10~min}
\end{mashsteps}

\begin{fermentationsteps}
\fermentationstep{\ftoc{60}}{Fermentation slowdown}
\fermentationstep{\ftoc{75}}{Full attenuation}
\end{fermentationsteps}

\end{methodandtiming}

\recipebreak

\begin{ingredientsblock}

\begin{malts}
\malt{Dingemans Pilsen}{\lbtokg{9.7}}
\malt{\maltflakedwheat}{\oztokg{16}}
\malt{\maltweyermannacidulated}{\oztokg{7}}
\end{malts}

\begin{hops}
\hop{Dextrose}{}{}{\lbtokg{2.7}}
\hop{\hopmagnum}{14~\%}{90~min}{\oztog{0.6}}
\hop{\hopsaaz}{3.8~\%}{5~min}{\oztog{1}}
\hop{\hopstyriangolding}{5.4~\%}{5~min}{\oztog{1}}
\hop{\hopsaaz}{3.8~\%}{\whirl{}{}}{\oztog{1}}
\hop{\hopstyriangolding}{5.4~\%}{\whirl{}{}}{\oztog{1}}
\end{hops}

\singleyeast{Wyeast 1388}

\end{ingredientsblock}

\end{recipe}
