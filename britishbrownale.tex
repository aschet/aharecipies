\begin{recipe}{Amahl's English Brown Ale}
% -----------------------------------------------------------------------------

\begin{aboutblock}
Recipe by Amahl Turczyn. \sourceaha
\end{aboutblock}

\specifications{\stylebritishbrownale}{\galtol{5.5}}{1.050}{1.013}{5~\%}{25}{\srmtoebc{18}}{60~min}{}

\begin{methodandtiming}

\begin{mashsteps}
\mashstep{\ftoc{153}}{60~min}
\end{mashsteps}

\begin{fermentationsteps}
\fermentationstep{\ftoc{64}}{Until fermentation starts}
\fermentationstep{\ftoc{68}}{Free raise to}
\end{fermentationsteps}

\begin{directions}
Water adjustment: \waterprofile{45--90}{0}{37--57}{0--60}{77--80}{}. Adjust the mash
pH to 5.2--5.3.
\end{directions}

\end{methodandtiming}

\recipebreak

\begin{ingredientsblock}

\begin{malts}
\malt{Maris Otter}{\lbtokg{5.5}}
\malt{Munich}{\lbtokg{3}}
\malt{Crisp Light Crystal}{\oztog{8}}
\malt{Crisp Brown}{\oztog{8}}
\malt{Pale Chocolate}{\oztog{4}}
\end{malts}

\begin{hops}
\hop{\hopeastkentgolding}{5~\%}{60~min}{\oztog{1.5}}
\hop{\hopeastkentgolding}{5~\%}{10~min}{\oztog{0.25}}
\end{hops}

\singleyeast{White Labs WLP007 / White Labs WLP002}

\end{ingredientsblock}

\end{recipe}

\begin{recipe}{Coriandered Mild Brown Ale}
% -----------------------------------------------------------------------------

\begin{aboutblock}
Recipe by Charlie Papazian. Refreshing, moderately floral, and spicy, with notable
malt character. \sourcezymurgy{March / April 2006}
\end{aboutblock}

\specifications{\stylebritishbrownale}{\galtol{5}}{1.040}{1.010}{4~\%}{25}{\srmtoebc{15}}{60~min}{}

\begin{methodandtiming}

\begin{mashsteps}
\mashstep{\ftoc{132}}{30~min}
\mashstep{\ftoc{155}}{30~min}
\mashstep{\ftoc{167}}{Mash out}
\end{mashsteps}

\begin{fermentationsteps}
\fermentationstep{\ftoc{70}}{1~week}
\fermentationstep{\ftoc{55}}{Transfer to secondary; 1~week}
\end{fermentationsteps}

\end{methodandtiming}

\recipebreak

\begin{ingredientsblock}

\begin{malts}
\malt{Pilsner}{\lbtokg{5}}
\malt{Wheat}{\oztog{12}}
\malt{Caramel / Crystal 40 L}{\oztog{12}}
\malt{Chocolate}{\oztog{4}}
\end{malts}

\begin{hops}
\hop{\hopvanguard}{5~\%}{60~min}{\oztog{0.75}}
\hop{\hopglacier}{6~\%}{30~min}{\oztog{0.5}}
\hop{\hopmthood}{4~\%}{10~min}{\oztog{1}}
\hop{Cracked Coriander Seeds}{}{1~min}{\oztog{0.75}}
\hop{Irish Moss}{}{}{\oztog{0.25}}
\end{hops}

\singleyeast{English Ale}

\begin{twists}
\twist{Cracked Coriander Seeds}{Secondary}{\oztog{0.5}}
\end{twists}

\end{ingredientsblock}

\end{recipe}

Competition.
