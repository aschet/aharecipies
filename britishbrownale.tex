\begin{recipe}{Amahl's English Brown Ale}
% -----------------------------------------------------------------------------

\begin{aboutblock}
Recipe by Amahl Turczyn. \sourceaha
\end{aboutblock}

\specifications{\stylebritishbrownale}{\galtol{5.5}}{1.050}{1.013}{5~\%}{25}{\srmtoebc{18}}{60~min}{}

\begin{methodandtiming}

\begin{mashsteps}
\mashstep{\ftoc{153}}{60~min}
\end{mashsteps}

\begin{fermentationsteps}
\fermentationstep{\ftoc{64}}{Until fermentation starts}
\fermentationstep{\ftoc{68}}{Free raise to}
\end{fermentationsteps}

\begin{directions}
Water adjustment: \waterprofile{45--90}{0}{37--57}{0--60}{77--80}{}. Adjust the mash
pH to 5.2--5.3.
\end{directions}

\end{methodandtiming}

\recipebreak

\begin{ingredientsblock}

\begin{malts}
\malt{Maris Otter}{\lbtokg{5.5}}
\malt{Munich}{\lbtokg{3}}
\malt{Crisp Light Crystal}{\oztog{8}}
\malt{Crisp Brown}{\oztog{8}}
\malt{Pale Chocolate}{\oztog{4}}
\end{malts}

\begin{hops}
\hop{\hopeastkentgolding}{5~\%}{60~min}{\oztog{1.5}}
\hop{\hopeastkentgolding}{5~\%}{10~min}{\oztog{0.25}}
\end{hops}

\singleyeast{White Labs WLP007 / White Labs WLP002}

\end{ingredientsblock}

\end{recipe}

\begin{recipe}{Charlie's Brown Ale}
% -----------------------------------------------------------------------------

\begin{aboutblock}
Mike Formisano, in collaboration with New Belgium Brewing. Silver medal in the
2013 GABF Pro-Am Competition. \sourceaha
\end{aboutblock}

\specifications{\stylebritishbrownale}{\galtol{5.25}}{1.058}{1.018}{5.2~\%}{52}{\srmtoebc{19}}{60~min}{2}

\begin{methodandtiming}

\begin{mashsteps}

\mashstep{\ftoc{143}}{30~min}

\end{mashsteps}

\begin{fermentationsteps}
\fermentationstep{\ftoc{68}}{2~weeks}
\fermentationstep{\ftoc{38}}{Transfer to secondary; 2~weeks}
\end{fermentationsteps}

\malt{Dingemans Biscuit}{\oztog{8}}
\malt{Briess Special Roast}{\oztog{8}}
\malt{Chocolate}{\oztog{4}}
\malt{Briess Victory}{\oztog{4}}
\malt{Simpsons Golden Naked Oats}{\oztog{4}}
\end{malts}

\begin{hops}
\hop{\hopnorthernbrewer}{8~\%}{60~min}{\oztog{1}}
\hop{\hopwillamette}{4~\%}{30~min}{\oztog{1}}
\hop{\hopfuggle}{5~\%}{15~min}{\oztog{1}}
\hop{Whirlfloc Tablet}{}{15~min}{1}
\hop{Five Star Super Moss HB}{}{15~min}{\tsptog{0.25}}
\hop{\hopeastkentgolding}{6~\%}{5~min}{\oztog{1}}
\end{hops}

\singleyeast{White Labs WLP002}

\end{ingredientsblock}

\end{recipe}
