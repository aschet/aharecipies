\begin{recipe}{BCJP Study Group Mild}
% -----------------------------------------------------------------------------

\begin{aboutblock}

Beer during the 2019 National Homebrew Competition Final Round in Providence, RI.
\sourceaha
\end{aboutblock}

\specifications{\styledarkmild}{\galtol{5.25}}{1.039}{1.013}{3.4~\%}{17.3}{\srmtoebc{23}}{60~min}{1.8}

\begin{methodandtiming}

\begin{mashsteps}

\end{mashsteps}

\begin{fermentationsteps}

\fermentationstep{\ftoc{66}}{Free raise to}

\end{methodandtiming}

\begin{malts}
