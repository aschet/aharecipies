\begin{recipie}{Big \& Beautiful Bock}
% -----------------------------------------------------------------------------

\begin{aboutblock}
Hank Keller of Cypress, TX, member of the Foam Rangers, won a silver medal in
Category \#6: Strong European Lager with a Dark Dopplebock during the 2019
National Homebrew Competition Final Round in Providence, RI. Keller's Doppelbock
was chosen as the runner-up entry among 269 entries in the category.
\end{aboutblock}

\specifications{\styledopplebock}{\galtol{6}}{1.110}{1.032}{10.2~\%}{18}{}{}{}

\end{mashsteps}

\begin{fermentationsteps}
\fermentationstep{\ftoc{51}}{2~weeks}
\fermentationstep{\ftoc{65}--\ftoc{68}}{3~days}
\fermentationstep{\ftoc{40}}{Slowly lower to}
\end{fermentationsteps}

\begin{directions}
Pull first \galtol{2} of wort into small kettle and boil down over 2 hours to
get about \galtol{0.5}. In main pot, collect about \galtol{7} of wort, and
boil down to \galtol{5}. Combine both parts into main kettle and boil
for 10 minutes.
\end{directions}

\end{methodandtiming}
