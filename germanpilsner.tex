\begin{aboutblock}
Jerry Mitchell of O'Fallon, MO, member of the Garage Brewers Society, won a silver
medal in Category \#3: Pilsner with a German Pils during the 2019 National Homebrew
Competition Final Round in Providence, RI. Mitchell's German Pils was chosen as the
runner-up entry among 286 entries in the category.  \sourceaha

\specifications{\stylegermanpilsner}{\galtol{5.7}}{1.049}{1.009}{4.8~\%}{35.8}{\srmtoebc{3.2}}{90~min}{}

\begin{methodandtiming}
 
\begin{mashsteps}
\mashstep{\ftoc{148}}{60~min}
\end{mashsteps}

\begin{fermentationsteps}
\fermentationstep{\ftoc{50}}{4~days}
\fermentationstep{\ftoc{60}}{7~days}
\fermentationstep{\ftoc{33}}{Reduce to over 3~days}
\end{fermentationsteps}

\begin{directions}
Water adjustment: add 3.5~g each of calcium chloride and calcium sulphate to
reverse osmosis water. Lager at \ftoc{33} for 4 weeks.
\end{directions}

\end{methodandtiming}
