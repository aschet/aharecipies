\begin{aboutblock}
Jerry Mitchell of O'Fallon, MO, member of the Garage Brewers Society, won a silver medal
in Category \#3: Pilsner with a German Pils during the 2019 National Homebrew Competition
Final Round in Providence, RI. Mitchell's German Pils was chosen as the runner-up entry

\specifications{\stylegermanpilsner}{\galtol{5.7}}{1.049}{1.009}{4.8~\%}{35.8}{\srmtoebc{3.2}}{90~min}

\begin{methodandtiming}
 
\begin{mashsteps}
\mashstep{\ftoc{148}}{60~min}
\end{mashsteps}

\begin{fermentationsteps}
\fermentationstep{\ftoc{50}}{4~days}
\fermentationstep{\ftoc{60}}{7~days}
\fermentationstep{\ftoc{33}}{Lower to over 3~days}
\end{fermentationsteps}

\begin{directions}
Water profile: add 3.5~g each of Calcium Chloride and Gypsum to 12 gallons of RO water.
Keg, add \ce{CO2} at 8 PSI and then drop the temperature every 12 hours until temperature
reaches \ftoc{33}. Increase \ce{CO2} pressure to 13 lbs. Hold at this temp for 4 weeks.

\end{directions}

\end{methodandtiming}
