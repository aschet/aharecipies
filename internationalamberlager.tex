\begin{recipie}{Winnebago International Amber Lager}
% -----------------------------------------------------------------------------

\begin{aboutblock}
Sean McCambridge of Lake Winnebago, MO, won a gold medal in Category \#4: Amber
European Beer with an International Amber Lager during the 2019 National Homebrew
Competition Final Round in Providence, RI. McCambridge's Amber European Beer was
chosen as the best among 365 entries in the category. \sourceaha
\end{aboutblock}

\specifications{\styleinternationalamberlager}{\galtol{6.5}}{1.056}{1.010}{5.6~\%}{31}{\srmtoebc{9.7}}{60~min}{2.75}

\begin{methodandtiming}
 
\begin{mashsteps}
\mashstep{\ftoc{152}}{60~min}
\end{mashsteps}

\begin{fermentationsteps}
\fermentationstep{\ftoc{50}}{7~days}
\fermentationstep{\ftoc{68}}{Raise to over 3~days; until fully attenuated}
\end{fermentationsteps}

\begin{directions}
Fine with gelatin. Lager for at least 2 weeks before drinking.
\end{directions}

\end{methodandtiming}

\pagebreak

\end{ingredientsblock}
