\begin{recipe}{2015 Big Brew Recipe 3: Killer Kölsch}
% -----------------------------------------------------------------------------

\begin{aboutblock}
Recipe by Gordon Strong. \sourceaha
\end{aboutblock}

\specifications{\stylekolsch}{\galtol{5}}{1.046}{1.011}{4.6~\%}{16}{\srmtoebc{3}}{90~min}{}

\begin{methodandtiming}

\begin{mashsteps}
\mashstep{\ftoc{131}}{10~min}
\mashstep{\ftoc{145}}{45~min}
\mashstep{\ftoc{158}}{20~min}
\mashstep{\ftoc{168}}{10~min}
\end{mashsteps}

\begin{fermentationsteps}
\fermentationstep{\ftoc{58}}{4~days}
\fermentationstep{\ftoc{68}}{Full attenuation}
\fermentationstep{\ftoc{40}}{2~months}
\end{fermentationsteps}

\begin{directions}
Water adjustment: reverse osmosis water with \tsptoml{0.25} of 10~\% phosphoric acid and
\tsptog{1} calcium chloride in the mash.
\end{directions}

\end{methodandtiming}

\recipebreak

\begin{ingredientsblock}

\begin{malts}
\malt{\maltpilsner}{\lbtokg{8.5}}
\malt{\maltvienna}{\oztokg{3.1}}
\malt{Caravienne}{\oztokg{3.1}}
\end{malts}

\begin{hops}
\hop{\hopliberty}{4.5~\%}{\fwh}{\oztog{0.3}}
\hop{\hophallertaumittelfruh}{4.3~\%}{30~min}{\oztog{1}}
\hop{\hopcrystal}{3.5~\%}{5~min}{\oztog{0.3}}
\end{hops}

\singleyeast{White Labs WLP029 / Wyeast 2565}

\end{ingredientsblock}

\end{recipe}
