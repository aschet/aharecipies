\begin{recipe}{Birra Corina Saison} % rechecked
% -----------------------------------------------------------------------------

\begin{aboutblock}
Recipe by Luis Cuellar of Croton-On-Hudson, NY. Gold medal in Category 20:
Saison during the 2019 National Homebrew Competition in Providence, RI.
\sourceaha
\end{aboutblock}

\specifications{\stylesaison}{\galtol{3}}{1.043}{1.008}{4.6~\%}{31}{\srmtoebc{3.7}}{90~min}{2.5}

\begin{methodandtiming}
 
\begin{mashsteps}
\mashstep{\ftoc{156}}{60~min}
\mashstep{\ftoc{170}}{Mash out}
\end{mashsteps}

\begin{fermentationsteps}
\fermentationstep{\ftoc{64}}{5~days}
\fermentationstep{\ftoc{80}}{Free raise; 20~days}
\end{fermentationsteps}

\begin{directions}
Condition bottles war, at \ftoc{75} for 14 days, then cold condition for at
least 14 days before consumption.
\end{directions}

\end{methodandtiming}

\recipebreak

\begin{ingredientsblock}

\begin{malts}
\malt{Avangard Pilsner}{\lbtokg{3.42}}

\hop{Cracked Coriander Seeds}{}{\foh{}{}}{\oztog{0.07}}
\end{hops}
