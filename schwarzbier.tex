\begin{recipe}{May the Schwartz Be With You} % rechecked
% -----------------------------------------------------------------------------

\begin{aboutblock}
Recipe by Frank Taddeo of Royersford, PA. Gold medal in Category 5: Dark European
Lager during the 2019 National Homebrew Competition in Providence, RI. \sourceaha
\end{aboutblock}

\specifications{\styleschwarzbier}{\galtol{5.5}}{1.056}{1.015}{5.3~\%}{30}{\srmtoebc{28}}{90~min}{2}

\begin{methodandtiming}
 
\begin{mashsteps}
\mashstep{\ftoc{154}}{45~min}
\mashstep{\ftoc{170}}{Mash out}
\end{mashsteps}

\begin{fermentationsteps}
\fermentationstep{\ftoc{50}}{5~days}
\fermentationstep{\ftoc{65}}{Raise over 2~days; 2~days}
\fermentationstep{\ftoc{50}}{Reduction over 2~days}
\end{fermentationsteps}

\begin{directions}
Water adjustment: mash pH of 5.3. Add whirlpool hops at \ftoc{170}.
\end{directions}

\end{methodandtiming}

\recipebreak

\begin{ingredientsblock}

\begin{malts}
\malt{Weyermann Munich II}{\lbtokg{6}}
\malt{Pilsner}{\lbtokg{4}}
\malt{Chocolate}{\oztog{6}}
\malt{Caramel / Crystal 40 L}{\oztog{6}}
\malt{Roasted Barley}{\oztog{3.5}}
\malt{Weyermann CARAFA SPECIAL II}{\oztog{3.5}}
\malt{Acidulated}{\oztog{2}}
\end{malts}

\begin{hops}
\hop{\hophallertaumittelfruh}{4~\%}{60~min}{\oztog{1.4}}
\hop{\hophallertaumittelfruh}{4~\%}{20~min}{\oztog{0.5}}
\hop{\hophallertaumittelfruh}{4~\%}{\whirl{}{10~min}}{\oztog{0.5}}
\end{hops}

\singleyeast{Wyeast 2206}

\end{ingredientsblock}

\end{recipe}
