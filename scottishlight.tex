\sourceaha
\end{aboutblock}

\specifications{\stylescottishlight}{\galtol{6}}{1.034}{1.010}{3.3~\%}{17}{\srmtoebc{18}}{60~min}{}

\begin{methodandtiming}

\begin{mashsteps}

\end{mashsteps}

\end{hops}

\end{ingredientsblock}

\end{recipe}
