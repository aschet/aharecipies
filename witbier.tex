\hop{Dried Moroccan Orange Peel}{}{1~min}{\oztog{5}}
\hop{Cracked Coriander Seeds}{}{\whirl{}{}}{\oztog{5}}
\end{hops}

\singleyeast{Wyeast 3787 / White Labs WLP530}

\end{ingredientsblock}
