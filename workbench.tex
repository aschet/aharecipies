\begin{aboutblock}
Recipe by Greg Young of Roseville, CA. Gold medal in Category 2: Pale European
Beer during the 2018 National Homebrew Competition in Portland, OR.
\sourceaha
\end{aboutblock}

\specifications{\styleamericanlager}{\galtol{5}}{1.048}{1.009}{5.1~\%}{21}{\srmtoebc{3}}{90~min}{3}

\begin{methodandtiming}

\begin{mashsteps}
\mashstep{\ftoc{148}}{90~min}
\mashstep{\ftoc{168}}{10~min}
\end{mashsteps}

\begin{fermentationsteps}
\fermentationstep{\ftoc{48}}{Pitch}
\fermentationstep{\ftoc{51}}{Free raise; attenaution to \sgtop{1.017}}
\fermentationstep{\ftoc{60}}{Free raise; 7~days}
\fermentationstep{\ftoc{34}}{Reduction by \ctoc{3}/day; 2~days}
\end{fermentationsteps}

\begin{directions}
Water adjustment: mash pH of 5.2. Lager for 14 days.
\end{directions}

\end{methodandtiming}
