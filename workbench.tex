\begin{recipe}{Lost Art American Pale Ale} % rechecked
% -----------------------------------------------------------------------------

\begin{aboutblock}
Recipe by Aaron Vieira and Scott Windsor of Costa Mesa. Bronze medal in Category
10: American Pale Ale during the 2018 National Homebrew Competition in Portland, OR.
\sourceaha
\end{aboutblock}

\specifications{\styleamericanpaleale}{\galtol{5}}{1.058}{1.011}{6.2~\%}{40}{\srmtoebc{5.5}}{90~min}{}

\begin{methodandtiming}

\begin{mashsteps}
\mashstep{\ftoc{152}}{60~min}
\end{mashsteps}

\begin{fermentationsteps}
\fermentationstep{\ftoc{70}}{9~days}
\end{fermentationsteps}

\begin{directions}
Water adjustment: \waterprofile{50}{10}{5}{105}{45}{}; mash pH of 5.3. Add whirlpool hops
at \ftoc{150}.
\end{directions}

\end{methodandtiming}

\recipebreak

\begin{ingredientsblock}

\begin{malts}
\malt{Two-row}{\lbtokg{15}}
\malt{Munich}{\lbtokg{1}}
\malt{Dingemans Biscuit}{\lbtokg{0.25}}
\end{malts}

\begin{hops}
\hop{\hopmagnum}{12.1~\%}{60~min}{\oztog{0.5}}
\hop{\hopcitra}{14~\%}{5~min}{\oztog{1}}
\hop{\hopmosaic}{13~\%}{5~min}{\oztog{0.5}}
\hop{\hopcentennial}{8~\%}{5~min}{\oztog{0.25}}
\hop{\hopcitra}{14~\%}{\whirl{}{10~min}}{\oztog{1}}
\hop{\hopmosaic}{13~\%}{\whirl{}{10~min}}{\oztog{0.5}}
\hop{\hopcentennial}{8~\%}{\whirl{}{10~min}}{\oztog{0.25}}
\hop{\hopcitra}{14~\%}{\dryh{}{3~days}}{\oztog{2}}
\hop{\hopmosaic}{13~\%}{\dryh{}{3~days}}{\oztog{1}}
\hop{\hopcentennial}{8~\%}{\dryh{}{3~days}}{\oztog{0.5}}
\end{hops}

\singleyeast{White Labs WLP001}

\end{ingredientsblock}

\end{recipe}
