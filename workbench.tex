\begin{recipe}{John's American Lager} % rechecked
% -----------------------------------------------------------------------------

\begin{aboutblock}
Recipe by Greg Young of Roseville, CA. Gold medal in Category 1: Pale American
Lager during the 2018 National Homebrew Competition in Portland, OR.
\sourceaha
\end{aboutblock}

\specifications{\styleamericanlager}{\galtol{5}}{1.044}{1.008}{4.7~\%}{13}{\srmtoebc{2}}{60~min}{3}

\begin{methodandtiming}

\begin{mashsteps}
\mashstep{\ftoc{149}}{90~min}
\mashstep{\ftoc{168}}{10~min}
\end{mashsteps}

\begin{fermentationsteps}
\fermentationstep{\ftoc{45}}{Pitch}
\fermentationstep{\ftoc{50}}{Free raise; attenaution to \sgtop{1.016}}
\fermentationstep{\ftoc{60}}{Free raise; 7~days}
\fermentationstep{\ftoc{32}}{Reduction by \ctoc{3}/day; 2~days}
\end{fermentationsteps}

\begin{directions}
Water adjustment: \waterprofile{50}{--}{--}{64}{80}{}, mash pH of 5.2. Lager for 21 days. 
\end{directions}

\end{methodandtiming}

\recipebreak

\begin{ingredientsblock}

\begin{malts}
\malt{Rahr Standard Two-row}{\lbtokg{6}}
\malt{Briess Flaked Maize}{\lbtokg{1.71}}
\malt{Briess Flaked Rice}{\oztokg{13.7}}
\end{malts}

\begin{hops}
\hop{\hophallertautradition}{6.6~\%}{60~min}{\oztog{0.5}}
\hop{\hophallertautradition}{6.6~\%}{5~min}{\oztog{0.5}}
\end{hops}

\singleyeast{White Labs WLP840}

\end{ingredientsblock}

\end{recipe}
