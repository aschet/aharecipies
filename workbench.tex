\begin{recipe}{Brew Age: Vienna Lager}
% -----------------------------------------------------------------------------

\begin{aboutblock}
Recipe by Johannes Kugler of Brew Age. A non traditional, late-hopped
Vienna Lager optimized for drinkability. Source: Krennmair, Andreas. Vienna Lager. Berlin, 2020.
\end{aboutblock}

\specifications{\styleviennalager}{\ltol{20}}{1.048}{1.012}{5~\%}{25}{\srmtoebc{5.4}}{70~min}{}

\begin{methodandtiming}

\begin{mashsteps}
\mashstep{\ctoc{50}}{5~min}
\mashstep{\ctoc{63}}{30~min}
\mashstep{\ctoc{73}}{30~min}
\mashstep{\ctoc{78}}{Mash out}
\end{mashsteps}

\begin{fermentationsteps}
\fermentationstep{\ctoc{7}}{Pitch}
\fermentationstep{\ctoc{9}}{Attenuation to 60~\%}
\fermentationstep{\ctoc{11}}{Full attenuation}
\end{fermentationsteps}

\end{methodandtiming}

\recipebreak

\begin{ingredientsblock}

\begin{malts}
\malt{\maltvienna}{\kgtokg{3.6}}
\malt{\maltpilsner}{\kgtokg{0.65}}
\malt{\maltweyermanncarared}{\kgtokg{0.23}}
\end{malts}

\begin{hops}
\hop{\hophallertaumittelfruh}{4~\%}{\fwh}{\gtog{9}}
\hop{\hopmagnum}{13~\%}{50~min}{\gtog{8}}
\hop{\hophallertaumittelfruh}{4~\%}{\whirl{}{30~min}}{\gtog{44}}
\end{hops}

\singleyeast{Fermentis SafLager W-34/70}

\end{ingredientsblock}

\end{recipe}
