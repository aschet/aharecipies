\begin{recipe}{Newcastle Brown Ale Clone}
% -----------------------------------------------------------------------------

\begin{aboutblock}
Recipe by Amahl Turczyn. Newcastle Brown Ale is a blend of two beers: a strong,
dark ale that's been aged with a younger, amber ale.
\sourcezymurgy{July / August 2001}
\end{aboutblock}

\specifications{\stylebritishbrownale}{\galtol{10}}{1.043}{1.010}{4.33~\%}{27}{\srmtoebc{23}}{90~min}{}

\begin{methodandtiming}

\begin{mashsteps}
\mashstep{\ftoc{154}}{60~min}
\end{mashsteps}

\begin{fermentationsteps}
\fermentationstep{\ftoc{64}}{}
\end{fermentationsteps}

\begin{directions}
Sparge into two kettles, one at a time, then adjust wort gravity to \sgtop{1.050}
and \sgtop{1.035}. Perform the boil and fermentation for each wort and blend the
resulting ales 50/50.
\end{directions}

\end{methodandtiming}

\recipebreak
