\begin{recipie}{Veteran's Day IPA}
% -----------------------------------------------------------------------------

\begin{aboutblock}
Stephen Cheng of Burke, VA, won a silver medal in Category \#16: American IPA
during the 2019 National Homebrew Competition Final Round in Providence, RI.
Cheng's American IPA was chosen as the runner-up entry among 353 entries in the
category. \sourceaha
\end{aboutblock}

\specifications{\styleamericanipa}{\galtol{5}}{1.053}{1.012}{5.46~\%}{81}{}{60~min}{2.25}

\begin{methodandtiming}
 
\begin{mashsteps}
\mashstep{\ftoc{150}}{60~min}
\end{mashsteps}

\begin{fermentationsteps}
\fermentationstep{\ftoc{70}}{}
\end{fermentationsteps}

\begin{directions}
On day 7, add dry hops. After two days cold crash, fine with gelatin, cold crash 2
more days.
\end{directions}

\end{methodandtiming}
